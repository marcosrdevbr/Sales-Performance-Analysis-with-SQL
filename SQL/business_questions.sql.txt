-- Question 1
SELECT SUM(Sales) AS Total_Revenue
FROM sales_data;

-- Question 2
SELECT
    Region,
    SUM(Sales) AS Revenue
FROM sales_data
GROUP BY Region
ORDER BY Revenue DESC;

-- Question 3
SELECT
    Product,
    SUM(Quantity) AS Units_Sold
FROM sales_data
GROUP BY Product
ORDER BY Units_Sold DESC;

-- Question 4
SELECT
    Category,
    SUM(Sales - Cost) AS Profit
FROM sales_data
GROUP BY Category
ORDER BY Profit DESC;

-- Question 5
SELECT
    AVG(Sales) AS Average_Order_Value
FROM sales_data;

-- Question 6
SELECT
    SUBSTR(Date, 1, 7) AS Month,
    SUM(Sales) AS Revenue
FROM sales_data
GROUP BY Month
ORDER BY Revenue DESC;
